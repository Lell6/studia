section .text
    global _start

section .data
    info_1 db 'Podaj znak: ', 0
    len_info_1 equ $-info_1
    newline db 10

section .bss
    one_str resb 1
    one_int resb 3
    result_str resb 10

section .text
_start:
    mov     eax, 4
    mov     ebx, 1
    mov     ecx, info_1
    mov     edx, len_info_1-1
    int     0x80

    mov     eax, 3
    mov     ebx, 0
    mov     ecx, one_str
    mov     edx, 1
    int     0x80   
    
    mov     eax, dword[one_str]
    sub     eax, '0'
    
    add     eax, 48
    mov     [one_int], eax
    
int_to_str:
    mov     eax, [one_int]   
    mov     ebx, 10         
    mov     ecx, 10            

    mov     edi, result_str    
    add     edi, 10         
    mov     byte [edi], 0     

convert_loop_str:
    dec     edi               
    xor     edx, edx         
    div     ebx                 

    add     dl, '0'           
    mov     [edi], dl       

    test    eax, eax           
    jnz     convert_loop_str   

    mov     [result_str], edi
    
    mov     eax, 4
    mov     ebx, 1
    mov     ecx, [result_str]
    mov     edx, 3
    int     0x80   

    mov eax, 1
    xor ebx, ebx
    int 0x80
